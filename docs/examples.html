Examples

Real-world examples of using CiteKit in your applications.

Python Examples

1. Basic PDF Ingestion

import asyncio
from citekit import CiteKitClient

async def main():
    client = CiteKitClient()
    
    # Ingest a research paper
    resource_map = await client.ingest(
        resource_path="research.pdf",
        resource_type="document"
    )
    
    # Print all discovered nodes
    print(f"Title: {resource_map.title}")
    for node in resource_map.nodes:
        print(f"  [{node.type}] {node.id}: {node.title}")
        print(f"    Pages: {node.location.pages}")

if __name__ == "__main__":
    asyncio.run(main())

3. Video Processing

import asyncio
from citekit import CiteKitClient

async def main():
    client = CiteKitClient()
    
    # Ingest a tutorial video
    resource_map = await client.ingest(
        resource_path="tutorial.mp4",
        resource_type="video"
    )
    
    # List all segments
    for node in resource_map.nodes:
        start = node.location.start
        end = node.location.end
        print(f"{node.id}: {start}s - {end}s ({node.title})")
    
    # Extract a specific clip
    evidence = await client.resolve("tutorial", "intro")
    print(f"Clip saved: {evidence.output_path}")

asyncio.run(main())

// Ingest all PDFs in a directory

3. Integration with LLM

import { CiteKitClient } from 'citekit';
import { GoogleGenerativeAI } from '@google/generative-ai';
import { readFileSync } from 'fs';

const citekit = new CiteKitClient();
const genai = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);
const model = genai.getGenerativeModel({ model: 'gemini-2.0-flash' });

// 1. Get specific content
const evidence = await citekit.resolve('research', 'methodology');

// 2. Read the mini-PDF
const pdfBytes = readFileSync(evidence.output_path);

// 3. Send to LLM
const result = await model.generateContent([
    { text: 'Summarize this methodology section:' },
    {
        inlineData: {
            mimeType: 'application/pdf',
            data: pdfBytes.toString('base64')
        }
    }
]);

console.log(result.response.text());

CLI Examples

Basic Workflow

# Ingest a document
citekit ingest my_document.pdf

# List all resources
citekit list

# Get the structure
citekit structure my_document

# Resolve a specific node
citekit resolve my_document.section1.intro

# Output: .citekit_output/my_document_pages_1_2.pdf

Advanced Usage

# Ingest with explicit type
citekit ingest --type video tutorial.mp4

# Ingest multiple files
citekit ingest *.pdf

# Specify custom API key
GEMINI_API_KEY=your_key citekit ingest doc.pdf

Research Assistant

Build an AI assistant that can reference specific sections of research papers.
